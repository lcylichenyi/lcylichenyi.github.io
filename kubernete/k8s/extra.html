# 如何移除节点

# 在master上运行
#封锁node，排干node上的pod
kubectl drain <node名称> --delete-local-data --force --ignore-daemonsets
# 等待状态变成SchedulingDisabled   
kubectl get nodes
# 删除节点
kubectl delete node <node名称>

# 在节点上运行
# 脱离master
kubeadm reset
